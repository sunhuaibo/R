``` r
df <- phyDat(t(df), type = "USER", levels = c(1, 0))
dm <- dist.hamming(df)
tree <- NJ(dm)
treeRatchet <- pratchet(df, start=tree, maxit=100,
                        minit=5, k=5, trace=0)
treeRatchet <- acctran(treeRatchet, df)
plot(treeRatchet, type = "unrooted")
```
